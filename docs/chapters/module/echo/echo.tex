\LevelOne{echo}
This module provides commands to send echo to players.

\LevelTwo{Sub-Module}
\LevelThree{/send-message}
\LevelThree{/send-broadcast}
\LevelThree{/send-actionbar}
\LevelThree{/send-title}
\begin{example}{Send title to a player}
    \ic{/send-title \player --mainTitle "<rainbow>Hello" --subTitle "<blue>World" --fadeInTicks 60 --stayTicks 60 --fadeOutTicks 60}
\end{example}

\begin{example}{Send title to online players}
    \ic{/foreach send-title \ph{player:name} --mainTitle "<rainbow>Hello \ph{player:name}"}
\end{example}
\LevelThree{/send-toast}

\begin{example}{Send toast to a player}
    \ic{/send-toast \player --icon minecraft:golden\_carrot <rb>eat this carrot}
\end{example}

\LevelThree{/send-chat}

\begin{example}{Send chat as a player}
    \ic{/send-chat Steve i am steve.}
\end{example}

\begin{example}{Send chat as a player for online players}
    \ic{/foreach send-chat \ph{player:name} i am \ph{player:name}}
\end{example}

\LevelThree{/send-bossbar}

\begin{example}{All in one exapmle}
    \ic{/send-bossbar \player --stepType BACKWARD --totalMs 5000 --color PURPLE --style NOTCHED\_6 --notifyMeOnComplete true --commandList "say the player \ph{player:name} is healed|heal \ph{player:name}" <rb>Healing is coming [elapsed\_time]/[total\_time]/[left\_time]}
\end{example}
